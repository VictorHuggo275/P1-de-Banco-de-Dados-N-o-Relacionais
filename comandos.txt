# COMANDOS MONGODB + DOCKER

## 1. DOCKER

# Cria e inicia um container MongoDB

docker run -d --name mongo-escola -p 27017:27017 mongo

# Lista os containers em execução

docker ps

# Acessa o terminal do MongoDB (mongosh)

docker exec -it mongo-escola mongosh

---

## 2. BANCO DE DADOS

# Cria ou seleciona o banco "escola"

use escola

# Cria a collection "alunos"

db.createCollection("alunos")

---

## 3. INSERÇÃO DE DADOS

db.alunos.insertMany([

---

## 4. CONSULTAS

# Retorna todos os alunos

db.alunos.find()

# Retorna alunos do curso ADS

db.alunos.find({ curso: "ADS" })

# Retorna alunos com idade maior que 21

db.alunos.find({ idade: { $gt: 21 } })

---

## 5. ATUALIZAÇÕES

# Atualiza a idade de um aluno

db.alunos.updateOne(
{ nome: "Rodrigo Pereira" },
{ $set: { idade: 24 } }
)

# Adiciona uma nova nota ao array de notas

db.alunos.updateOne(
{ nome: "Suellen Souza" },
{ $push: { notas: 10 } }
)

---

## 6. REMOÇÃO

# Remove um aluno da collection

db.alunos.deleteOne({ nome: "Ramona Santos" })

---

## 7. AGGREGATIONS

# Calcula a média das notas por aluno

db.alunos.aggregate([
{
$project: {
nome: 1,
media: { $avg: "$notas" }
}
}
])

# Conta a quantidade de alunos por curso

db.alunos.aggregate([
{
$group: {
_id: "$curso",
quantidade: { $sum: 1 }
}
}
])
